country_event = {
	id = submod_custom_events.2
	title = submod_custom_events.2.t
	desc = submod_custom_events.2.desc
	picture = MAGIC_STUDY_AND_RELAX_eventPicture	
	
	trigger = {
		court_mage = 3
		#NOT = { has_country_flag = court_mage }	#BomjSBagetom Anbennar Submod
		OR = {
			NOT = { has_country_flag = court_mage }
			if = {	#BomjSBagetom Anbennar Submod
				limit = {
					administrative_ideas = 7
				}
				had_country_flag = {
					flag = court_mage
					days = 18250
				}
			}
			else = {
				had_country_flag = {
					flag = court_mage
					days = 36500
				}
			}
		}
	}
	
	mean_time_to_happen = {
		months = 300
		
		
		modifier = {
			factor = 0.9
			innovativeness_ideas = 4
		}
		modifier = {
			factor = 0.9
			innovativeness_ideas = 6
		}
		modifier = {
			factor = 1.1
			NOT = { innovativeness_ideas = 4 }
		}
		modifier = {
			factor = 1.1
			NOT = { innovativeness_ideas = 2 }
		}
	}
	
	option = {
		name = submod_custom_events.2.a	# His efforts will be admired throughout the nation
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				stability = 2
			}
		}
		#add_meritocracy_effect = yes
		add_stability_or_adm_power = yes
		if = {
			limit = {
				administrative_ideas = 7
			}
			add_country_modifier = {
				name = "court_mage_boost"
				duration = 3650
			}
		}
		else = {
			add_country_modifier = {
				name = "court_mage_boost"
				duration = 2190
			}
		}
		set_country_flag = court_mage
	}
		
	option = {
		name = submod_custom_events.2.b	# His reputation will reach further than our borders
		ai_chance = {
			factor = 50
			modifier = {
				factor = 0
				NOT = { stability = 1 }
			}
		}
		#add_meritocracy_effect = yes
		add_prestige = 50
		if = {
			limit = {
				administrative_ideas = 7
			}
			add_country_modifier = {
				name = "court_mage_boost"
				duration = 3650
			}
		}
		else = {
			add_country_modifier = {
				name = "court_mage_boost"
				duration = 2190
			}
		}
		set_country_flag = court_mage
	}
}
